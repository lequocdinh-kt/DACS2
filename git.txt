1. lấy về 
    git pull
    
2. đẩy lên
    git add --all
    git commit -m "message"
    git push -u origin main
    
3. tạo nhánh mới
    git checkout -b <tên nhánh mới>
    git push -u origin <tên nhánh mới>

4. gộp nhánh
    git checkout main
    git merge <tên nhánh mới>
    git push -u origin main    (đẩy lên phải có add và commit)
    
5. xóa nhánh
    git branch -d <tên nhánh mới>
    git push origin --delete <tên nhánh mới>
6. xem trạng thái
    git status
7. xem lịch sử
    git log
8. xem sự khác biệt
    git diff
9. khôi phục file
    git checkout -- <tên file>
10. xem các nhánh
    git branch
11. chuyển nhánh
    git switch <tên nhánh>
